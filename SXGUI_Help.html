2， SXGUI_CreateRoot：

        该函数为创建根目录的函数，在 SXGUI_Init 中已经调用，无需用户再次调用；

xxxxxxxxxx
int SXGUI_CreateRoot(char* Name, void (*Interface)(INTERFACE_PARAMETERS)){


  if(ROOT != NULL) {
    return SXGUI_ERROR;     //如果已创建
  }
  ROOT = malloc(sizeof(SXGUI_MenuItem));
  ROOT->name = Name;
  ROOT->Interface = Interface;
  return SXGUI_OK;
}

其中的 ROOT 是全局的指针变量，定义如下：

6，SXGUI_DeleteMenu：

该函数功能为从链表中删除并释放菜单项，不会影响其他菜单项；

xxxxxxxxxx
void SXGUI_DeleteMenu(SXGUI_MenuItem* Menu) {
  if(Menu->Pre == NULL) {
    Menu->Parent->Sub = Menu->Next;
    Menu->Next->Pre = NULL;
  } else {
    Menu->Pre->Next = Menu->Next;
    if(Menu->Next != NULL) {
      Menu->Next->Pre = Menu->Pre;
    }
  }
  free(Menu);
}

8，SXGUI_RootAddSubMenu：

该函数功能为在根目录下添加菜单项；

xxxxxxxxxx
int SXGUI_RootAddSubMenu(SXGUI_MenuItem* SubMenu) {
  if(SubMenu->Parent != NULL) { //如果已被添加
    return SXGUI_ERROR;
  }


  SubMenu->Parent = ROOT;
  SXGUI_MenuItem* Ptr = ROOT->Sub;


  if(Ptr != NULL) {
    for(; Ptr->Next != NULL; Ptr = Ptr->Next);
    Ptr->Next = SubMenu;
  } else {
    ROOT->Sub = SubMenu;
  }


  SubMenu->Pre = Ptr;


  return SXGUI_OK;
}
